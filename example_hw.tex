\begin{solution}

\begin{problem}
Solve the system of equations:
\[
\begin{cases}
x + y = 3 \\
2x - y = 0
\end{cases}
\]
\end{problem}

\begin{solution}
Adding the two equations: $3x = 3$, so $x = 1$. Substituting back, $y = 2$.
\end{solution}
